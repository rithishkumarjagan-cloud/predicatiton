  const chartData = {
    '1m': {
      labels: ['May 12','May 15','May 19','May 22','May 26','May 29','Jun 3','Jun 6','Jun 10','Jun 10*'],
      actual:    [2710, 2730, 2760, 2780, 2800, 2815, 2820, 2830, 2847, null],
      predicted: [null, null, null, null, null, null, null, null, 2847, 2891]
    },
    '3m': {
      labels: ['Apr 1','Apr 15','May 1','May 15','Jun 1','Jun 10','Jun 10*'],
      actual:    [2580, 2620, 2680, 2740, 2800, 2847, null],
      predicted: [null, null, null, null, null, 2847, 2920]
    },
    '6m': {
      labels: ['Jan','Feb','Mar','Apr','May','Jun','Jun*'],
      actual:    [2300, 2400, 2500, 2620, 2740, 2847, null],
      predicted: [null, null, null, null, null, 2847, 2935]
    }
  };

  let myChart;
  function buildChart(range) {
    const d = chartData[range];
    const ctx = document.getElementById('stockChart').getContext('2d');
    if (myChart) myChart.destroy();
    myChart = new Chart(ctx, {
      type: 'line',
      data: {
        labels: d.labels,
        datasets: [
          {
            label: 'Actual Price (₹)',
            data: d.actual,
            borderColor: '#0f3460',
            backgroundColor: 'rgba(15,52,96,0.08)',
            borderWidth: 2.5,
            pointRadius: 4,
            fill: true,
            tension: 0.35,
            spanGaps: false
          },
          {
            label: 'Predicted Price (₹)',
            data: d.predicted,
            borderColor: '#e94560',
            borderDash: [7, 4],
            borderWidth: 2.5,
            pointRadius: 6,
            pointBackgroundColor: '#e94560',
            fill: false,
            tension: 0.3,
            spanGaps: false
          }
        ]
      },
      options: {
        responsive: true,
        plugins: {
          legend: { labels: { font: { size: 12 } } },
          tooltip: { callbacks: { label: ctx => ' ₹' + ctx.parsed.y?.toLocaleString() } }
        },
        scales: {
          x: { grid: { display: false } },
          y: {
            grid: { color: 'rgba(0,0,0,0.05)' },
            ticks: { callback: v => '₹' + v.toLocaleString() }
          }
        }
      }
    });
  }

  function setRange(r, btn) {
    document.querySelectorAll('.range-btns button').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    buildChart(r);
  }

  buildChart('1m');

  function runPrediction() {
    const open = parseFloat(document.getElementById('openPrice').value) || 2820;
    const high = parseFloat(document.getElementById('highPrice').value) || 2860;
    const low  = parseFloat(document.getElementById('lowPrice').value)  || 2795;
    const vol  = parseFloat(document.getElementById('volume').value)    || 48;
    const model = document.getElementById('modelSel').value;
    const stock = document.getElementById('stockSel').value;

    const multipliers = { 'LSTM (Deep Learning)': 1.024, 'Random Forest': 1.019, 'Linear Regression': 1.013 };
    const confidence  = { 'LSTM (Deep Learning)': '91%', 'Random Forest': '86%', 'Linear Regression': '75%' };

    const avg = (open + high + low) / 3;
    const predicted = Math.round(avg * (multipliers[model] || 1.02) + vol * 0.3);

    document.getElementById('resultPrice').textContent = '₹' + predicted.toLocaleString();
    document.getElementById('resultMeta').textContent =
      `Model: ${model.split(' ')[0]} | Confidence: ${confidence[model]} | Stock: ${stock}`;
    document.getElementById('resultBox').style.display = 'block';
  }
</script>
